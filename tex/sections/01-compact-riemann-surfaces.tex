\section{Compact Riemann Surfaces}
\label{sec:compact-riemann-surfaces}

Throughout, \(X,Y,Z\) denote compact connected Riemann surfaces, and maps
between them are holomorphic unless otherwise specified. Compactness implies
that every holomorphic function \(X\to\C\) is constant. Holomorphic maps
between compact Riemann surfaces are either constant or finite branched
coverings.

\subsection{Order of vanishing}

Throughout, all charts are holomorphic. We work locally and then glue.

\begin{definition}[Order of vanishing of a meromorphic germ at a point]
\label{def:vanishing-order}
For a holomorphic chart \(\varphi:U\to\C\) sending \(p\mapsto 0\) and a
meromorphic germ \(f\) at \(p\), the order of vanishing
\(\ord_p(f)\in\Z\cup\{+\infty\}\) is the order of the Laurent series of
\(f\circ\varphi^{-1}\) at \(0\). The convention is
\(\ord_p(0)=+\infty\) and \(\ord_p(f)<0\) for poles. The value is
independent of the chart.
\depends{Mathlib's \texttt{analyticOrderAt} for normed spaces; chart
independence proved via composition of analytic equivalences.}
\notready
\end{definition}

\begin{proofsketch}[Lean formalization strategy]
Mathlib v4.28 provides \texttt{MeromorphicAt} and \texttt{meromorphicOrderAt}
in \texttt{Mathlib.Analysis.Meromorphic.\{Basic, Order\}} for functions
\(f : \mathbb{k} \to E\) on a normed field, so the manifold case requires a
project-side wrapper that reduces to the chart-local statement. Concrete
plan (vetted against v4.28 by an LLM cross-check round):

\begin{enumerate}
\item Define \texttt{MeromorphicAtX (f : X $\to$ \(\C\)) (p : X)} as
  \texttt{MeromorphicAt (f $\circ$ (extChartAt $\mathcal I(\C)$ p).symm)
  (extChartAt $\mathcal I(\C)$ p p)}, using the extended chart of
  \texttt{IsManifold $\mathcal I(\C)$ $\omega$ X}.
\item Define \texttt{orderAt p f hf := meromorphicOrderAt (f $\circ$
  (extChartAt $\mathcal I(\C)$ p).symm) (extChartAt $\mathcal I(\C)$ p p) :
  WithTop $\Z$}, with \texttt{hf : MeromorphicAtX f p}.
\item Chart-independence: for two extended charts
  \(e_1, e_2\) at \(p\), the transition map
  \(g := e_2 \circ e_1^{-1}\) is analytic at \(e_1(p)\) and
  \(g(e_1(p)) = e_2(p)\); the manifold typeclass provides
  \texttt{ContMDiff} of order \(\omega\), which on the standard model
  \(\mathcal I(\C)\) means analytic. The complex inverse function theorem
  gives \(g'(e_1(p)) \ne 0\). Apply
  \texttt{MeromorphicAt.meromorphicOrderAt\_comp}
  (\texttt{Mathlib.Analysis.Meromorphic.Order} line 725) to conclude
  \(\ord_{e_1}(f \circ e_1^{-1}) = \ord_{e_2}(f \circ e_2^{-1})\).
\end{enumerate}

Estimated cost: ~250--350 LOC across one new file
(\texttt{Jacobian/HolomorphicForms/VanishingOrder.lean}). The transition
derivative-non-zero step is the only piece without an off-the-shelf
Mathlib lemma; it costs ~30--50 LOC and reduces to the complex inverse
function theorem (\texttt{Mathlib.Analysis.Calculus.InverseFunctionTheorem}).
Naming note: \texttt{PartialHomeomorph}, not \texttt{LocalHomeomorph}
(already renamed in pre-v4.28).
\end{proofsketch}

\begin{lemma}[Discreteness of the divisor support]
\label{lem:divisor-discrete}
\uses{def:vanishing-order}
For a nonzero meromorphic function \(f\) on \(X\), the set
\(\{p\in X:\ord_p(f)\ne 0\}\) has no accumulation point.
\notready
\end{lemma}

\begin{proofsketch}
Locally, in a chart \(\varphi\), \(f\circ\varphi^{-1}\) is meromorphic on
an open subset of \(\C\). Either the Laurent series at \(\varphi(p)\)
vanishes identically (impossible for a nonzero meromorphic function on
a connected open set, by the identity theorem), or it has finite order.
Finite order at a point \(p\) gives a chart neighborhood on which \(f\)
has no zeros and no poles other than at \(p\). The identity theorem
applied across the chart atlas globalizes this.
\end{proofsketch}

\begin{lemma}[Compactness gives finite divisor support]
\label{lem:divisor-finite-support}
\uses{lem:divisor-discrete}
For a nonzero meromorphic function \(f\) on the compact Riemann surface
\(X\), the set \(\{p\in X:\ord_p(f)\ne 0\}\) is finite.
\notready
\end{lemma}

\begin{proof}
A subset of a compact space with no accumulation point is finite.
\end{proof}

\subsection{Divisor group}

\begin{definition}[Divisor group]
\label{def:divisor}
The divisor group \(\Div(X)\) is the free abelian group on the points of
\(X\). A divisor is a formal finite \(\Z\)-linear combination
\(D=\sum_{p}n_p\,p\) with \(n_p\in\Z\) and \(\{p:n_p\ne 0\}\) finite.
\depends{Mathlib's \texttt{FreeAbelianGroup}.}
\notready
\end{definition}

\begin{definition}[Degree of a divisor]
\label{def:divisor-degree}
\uses{def:divisor}
The degree map \(\deg:\Div(X)\to\Z\) sends \(D=\sum_p n_p\,p\) to
\(\sum_p n_p\). It is a group homomorphism.
\notready
\end{definition}

\begin{proofsketch}[Lean formalization strategy]
Mathlib v4.28 already has \texttt{FreeAbelianGroup} in
\texttt{Mathlib.GroupTheory.FreeAbelianGroup}, with the universal map
\texttt{FreeAbelianGroup.lift : ($\alpha\to\beta$) $\simeq$
(FreeAbelianGroup $\alpha\to_+\beta$)} and the simp lemma
\texttt{FreeAbelianGroup.lift\_apply\_of}. Cross-checked against v4.28 by
an LLM round (ChatGPT proposed the wrapper, Grok confirmed name
correctness against the v4.28 source and refined the instance pattern).
The two definitions reduce to a $\sim$15--25 LOC file
\texttt{Jacobian/HolomorphicForms/Divisor.lean}:
\begin{enumerate}
\item \texttt{def Divisor (X : Type u) : Type u := FreeAbelianGroup X},
  with \texttt{instance : AddCommGroup (Divisor X) := inferInstanceAs \_}.
  No \texttt{noncomputable}, no \texttt{@[reducible]}, no extra type
  classes on \(X\) (the blueprint says ``free abelian group on the
  points of \(X\)'' — \(X\) is a plain \texttt{Type}).
\item \texttt{def Divisor.of (p : X) : Divisor X := FreeAbelianGroup.of p}.
\item \texttt{def Divisor.degree : Divisor X $\to_+$ $\Z$ :=
  FreeAbelianGroup.lift (fun \_ => 1)}.
\item \texttt{@[simp] theorem degree\_of (p : X) :
  degree (of p) = 1 := by simp [degree, of]} —
  one-line proof via \texttt{FreeAbelianGroup.lift\_apply\_of}.
\end{enumerate}
No new sorries, no Mathlib gap. This is a clean leaf chosen specifically
to verify the cloud SessionStart cache-prime hook.
\end{proofsketch}

\subsection{Meromorphic functions, principal divisors, principal degree}

\begin{definition}[Meromorphic functions on a compact Riemann surface]
\label{def:meromorphic-function}
\uses{def:vanishing-order}
The set \(\Mer(X)\) of meromorphic functions on \(X\) is the set of
sections of the sheaf of meromorphic germs. \(\Mer(X)\) is a field with
the constants \(\C\) embedded as constant functions, and a nonzero
\(f\in\Mer(X)\) defines a holomorphic map \(X\to\CPone\) with poles at
\(\{p:\ord_p(f)<0\}\).
\depends{Mathlib has \texttt{MeromorphicAt} on normed spaces; gluing along a
chart atlas to a global field on \(X\) is project-side work.}
\notready
\end{definition}

\begin{definition}[Principal divisor of a meromorphic function]
\label{def:principal-divisor}
\uses{def:divisor,def:meromorphic-function,lem:divisor-finite-support}
For a nonzero \(f\in\Mer(X)\),
\[
  (f) := \sum_{p\in X} \ord_p(f)\,p \in \Div(X).
\]
The map \((\,\cdot\,):\Mer(X)^{\times}\to\Div(X)\) is a group
homomorphism (\(\Mer(X)^{\times}\) under multiplication).
\notready
\end{definition}

\begin{definition}[Principal divisors]
\label{def:principal-divisors}
\uses{def:principal-divisor}
The subgroup of principal divisors is
\(\Prin(X) := \{(f) : f\in\Mer(X)^{\times}\}\subset\Div(X)\).
\notready
\end{definition}

\begin{theorem}[Principal divisors have degree zero]
\label{thm:principal-degree-zero}
\uses{def:principal-divisor,def:divisor-degree}
For every nonzero \(f\in\Mer(X)\), \(\deg(f)=0\).
\notready
\end{theorem}

\begin{proofsketch}
The classical proof uses the argument principle on a polygonal model of
\(X\): the integral
\(\frac{1}{2\pi i}\oint_{\partial X}\,d\!\log f\) counts zeros minus
poles, but the polygon's boundary cancels out by the identification of
opposite sides, so the difference is zero. Equivalently, this is a
special case of \(\sum\Res = 0\) for the meromorphic 1-form \(df/f\).
\end{proofsketch}

\subsection{Meromorphic functions as maps to \texorpdfstring{\(\CPone\)}{CP1}}

\begin{definition}[Map to \(\CPone\) associated to a meromorphic function]
\label{def:meromorphic-to-cp1}
\uses{def:meromorphic-function}
For nonzero \(f\in\Mer(X)\), define \(\widehat{f}:X\to\CPone\) by
\(\widehat{f}(p)=f(p)\) where \(f\) is holomorphic at \(p\) and
\(\widehat{f}(p)=\infty\) at poles. This is a holomorphic map of compact
Riemann surfaces.
\notready
\end{definition}

\begin{theorem}[Meromorphic-to-\(\CPone\) bijection]
\label{thm:meromorphic-as-cp1-map}
\uses{def:meromorphic-to-cp1}
The map \(f\mapsto\widehat{f}\) is a bijection between nonzero
meromorphic functions on \(X\) and nonconstant holomorphic maps
\(X\to\CPone\).
\notready
\end{theorem}

\begin{proofsketch}
Forward: a meromorphic function on \(X\) is exactly a holomorphic
section of \(X\to\CPone\) defined off the polar set, with continuous
extension to \(\infty\) at the polar set; the extension is forced and
holomorphic. Reverse: given \(g:X\to\CPone\), restrict to
\(g^{-1}(\C)\) and observe the result is holomorphic, with the
preimages of \(\infty\) becoming poles.
\end{proofsketch}

\subsection{Riemann-Roch space}

\begin{definition}[Riemann-Roch space \(L(D)\)]
\label{def:riemann-roch-space}
\uses{def:divisor,def:meromorphic-function,def:principal-divisor}
For a divisor \(D\),
\[
  L(D) := \{f\in\Mer(X) : (f)+D\ge 0\}\cup\{0\}.
\]
\notready
\end{definition}

\begin{lemma}[\(L(D)\) is a complex vector space]
\label{lem:riemann-roch-space-vector}
\uses{def:riemann-roch-space}
\(L(D)\) is a \(\C\)-vector subspace of \(\Mer(X)\).
\notready
\end{lemma}

\begin{formalizationnote}
The repository currently has substantial chart and path-integration
infrastructure, but not a full divisor or meromorphic-function theory on
compact Riemann surfaces. The pieces above are individually
project-sized but each is a clean stand-alone obligation; some
(\texttt{def:divisor}, \texttt{def:divisor-degree}) reduce to existing
Mathlib infrastructure (\texttt{FreeAbelianGroup}). Others
(\texttt{def:meromorphic-function}, \texttt{thm:principal-degree-zero})
require sheaf-of-meromorphic-germs gluing or argument-principle work
that is not yet in Mathlib.
\end{formalizationnote}
