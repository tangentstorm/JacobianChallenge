% \code{Mathlib.Long.Identifier.Path} renders in monospace and may
% break across lines at every '.' or '_'.  Without the discretionary
% breakpoints, long Mathlib identifier paths (e.g.\
% Mathlib.Analysis.InnerProductSpace.Spectrum.compactSelfAdjoint)
% overflow the right margin -- visible as overfull \hbox warnings
% during pdflatex and as clipped names in the rendered PDF.
%
% The substitution chain runs xstring's \StrSubstitute over
% \detokenize'd input.  xstring is purely macro-recursive (no
% \futurelet) so it composes with \detokenize -- unlike seqsplit,
% which fails because its \futurelet can't traverse detokenize's
% catcode-12 token stream.
\newcommand{\code}[1]{%
  \texttt{%
    \StrSubstitute{\detokenize{#1}}{.}{.\linebreak[1]}%
  }%
}
